561  {
562  if (n_threads == 0) {

564  }
565  const std::size_t n_target = threads_.size() + n_threads;
566  while (threads_.size() < n_target) {

575  threads_.push_back(std::move(thread));

597  functors_.push(functor);
598  }
599  cond_var_.notify_one();
600  }

604  void worker() {
605  for (;;) {
606  std::function<void()> functor;
607  {
608  std::unique_lock<std::mutex> lock{mutex_};
609  cond_var_.wait(lock, [this]{
610  return done_ || !functors_.empty();
611  });
612  if (done_ && functors_.empty()) {
613  break;
614  }
615  functor = functors_.front();
616  functors_.pop();
617  }
618  functor();
619  }
620  }

622  void shutdown() {
623  {
624  std::lock_guard<std::mutex> lock{mutex_};
625  done_ = true;
626  }
627  cond_var_.notify_all();
628  for (std::thread& thread : threads_) {
629  thread.join();
630  }
631  threads_.clear();
632  }
633 
634  bool done_ = false;
635  std::vector<std::thread> threads_;
636  std::queue<std::function<void()>> functors_;
637  std::condition_variable cond_var_;

1524 /// A simple circular buffer (FIFO).
1525 /// ValueType must support default construction. The buffer lets you push
1526 /// new values onto the back and pop old values off the front.
1527 template<typename ValueType>
1528 class circular_buffer {
1529 public:
1530 
1531  static_assert(std::is_default_constructible_v<ValueType>, "ValueType must be default constructible");
1532 
1533  using value_type = ValueType;
1534 
1535  /// Constructs a circular buffer with a given fixed capacity
1536  explicit
1537  circular_buffer(std::size_t capacity)
1538  : data_(capacity)
1539  {
1540  if (capacity < 1) {
1541  throw transwarp::invalid_parameter{"capacity"};
1542  }
1543  }
1544 
1545  // delete copy/move semantics
1546  circular_buffer(const circular_buffer&) = delete;

1548  circular_buffer(circular_buffer&& other) = delete;
1549  circular_buffer& operator=(circular_buffer&&) = delete;

1552  /// of the buffer then the oldest value gets dropped (the one at the front).
1553  template<typename T, typename = std::enable_if_t<std::is_same_v<std::decay_t<T>, value_type>>>
1554  void push(T&& value) {
1555  data_[end_] = std::forward<T>(value);
1556  increment();
1557  }

1560  /// This is undefined if the buffer is empty

1562  return data_[front_];
1563  }
1564

1566  void pop() {
1567  if (!empty()) {
1568  data_[front_] = ValueType{};
1569  decrement();
1570  }
1571  }
1572

1574  std::size_t capacity() const {
1575  return data_.size();
1576  }
1577

1579  /// equals the capacity of the buffer
1580  std::size_t size() const {
1581  return size_;
1582  }
1583

1585  bool empty() const {
1586  return size_ == 0;
1587  }
1588

1590  bool full() const {
1591  return size_ == data_.size();
1592  }

1595  void swap(circular_buffer& buffer) {
1596  std::swap(end_, buffer.end_);
1597  std::swap(front_, buffer.front_);

1603

1607  } else {
1608  ++value;
1609  }
1610  }
1611

1614  if (full()) {
